Installation

# Clone the repository
git clone https://github.com/bekingdomcomejoker-cpu/omega-warfare-analytics.git
cd omega-warfare-analytics

# Install dependencies
pnpm install

# Set up environment variables
cp .env.example .env.local

# Run database migrations
pnpm run db:migrate

# Create Primary Researcher account
pnpm run create:researcher

# Start development server
pnpm run dev
